39 template<typename T, std::size_t PoolSize, PoolAllocStrategy Strategy>
40 struct DefaultPoolAllocator {
41  DefaultPoolAllocatorData<Strategy> data;
42 
43  explicit DefaultPoolAllocator(std::invocable auto &&generator, std::list<std::unique_ptr<T>> &available,
44  std::list<std::unique_ptr<T>> &inUse)
45  : generator(generator) {
46  if constexpr (Strategy == PoolAllocStrategy::Preallocate) { alloc_n(available, inUse, PoolSize); }
47  }
48  void on_lease(std::list<std::unique_ptr<T>> &available, std::list<std::unique_ptr<T>> &inUse) {
49  if constexpr (Strategy == PoolAllocStrategy::IncreaseBy2x) {
50  if (data.capacity_ == inUse.size() && data.capacity_ < PoolSize) {
51  const auto n_to_alloc = std::max(std::min(data.capacity_ * 2, PoolSize - data.capacity_), 1ull);
52  alloc_n(available, inUse, n_to_alloc);
53  data.capacity_ += n_to_alloc;
54  }
55  } else if constexpr (Strategy == PoolAllocStrategy::OnDemand) {
56  if (data.capacity_ == inUse.size() && data.capacity_ < PoolSize) {
57  alloc_n(available, inUse, 1);
58  ++data.capacity_;
59  }
60  }
61  }
62  void on_release(std::list<std::unique_ptr<T>> &available, std::list<std::unique_ptr<T>> &inUse) {}
63 
64  std::size_t capacity() { return data.capacity_; }
65 
66  void shrink(std::list<std::unique_ptr<T>> &available, std::list<std::unique_ptr<T>> &inUse) {
67  const auto remove_cnt = inUse.size();
68  inUse.clear();
69  data.capacity_ -= remove_cnt;
70  }
71 
72  void alloc_n(std::list<std::unique_ptr<T>> &available, std::list<std::unique_ptr<T>> &inUse, std::size_t n) {
73  std::generate_n(std::back_inserter(available), n, [&] { return std::make_unique<T>(generator()); });
74  }
75 
76  std::function<T()> generator;
77 };

117  [[nodiscard]] pointer lease() {
118  std::unique_lock lock{mutex};
119  allocator.on_lease(available_, inUse);
120  if (available_.empty()) { throw std::runtime_error{"Pool has no available objects."}; }
121  const auto first_available_iter = available_.begin();
122  auto &ref = inUse.emplace_front(std::move(*first_available_iter));
123  available_.erase(first_available_iter);
124  return std::experimental::make_observer(ref.get());
125  }
126 
131  void release(pointer object) {
132  std::unique_lock lock{mutex};
133  if (auto iter =
134  std::find_if(inUse.begin(), inUse.end(), [&object](const auto &obj) { return obj.get() == object.get(); });
135  iter != inUse.end()) {
136  auto &ref = available_.emplace_back(std::move(*iter));
137  inUse.erase(iter);
138  allocator.on_release(available_, inUse);
139  }
140  }
